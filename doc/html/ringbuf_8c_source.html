27 /*
28  * Atomic multi-producer single-consumer ring buffer, which supports
29  * contiguous range operations and which can be conveniently used for
30  * message passing.
31  *
32  * There are three offsets -- think of clock hands:
33  * - NEXT: marks the beginning of the available space,
34  * - WRITTEN: the point up to which the data is actually written.
35  * - Observed READY: point up to which data is ready to be written.
36  *
37  * Producers
38  *
39  * Observe and save the 'next' offset, then request N bytes from
40  * the ring buffer by atomically advancing the 'next' offset. Once
41  * the data is written into the "reserved" buffer space, the thread
42  * clears the saved value; these observed values are used to compute
43  * the 'ready' offset.
44  *
45  * Consumer
46  *
47  * Writes the data between 'written' and 'ready' offsets and updates
48  * the 'written' value. The consumer thread scans for the lowest
49  * seen value by the producers.
50  *
51  * Key invariant
52  *
53  * Producers cannot go beyond the 'written' offset; producers are
54  * also not allowed to catch up with the consumer. Only the consumer
55  * is allowed to catch up with the producer i.e. set the 'written'
56  * offset to be equal to the 'next' offset.
57  *
58  * Wrap-around
59  *
60  * If the producer cannot acquire the requested length due to little
61  * available space at the end of the buffer, then it will wraparound.
62  * WRAP_LOCK_BIT in 'next' offset is used to lock the 'end' offset.
63  *
64  * There is an ABA problem if one producer stalls while a pair of
65  * producer and consumer would both successfully wrap-around and set
66  * the 'next' offset to the stale value of the first producer, thus
67  * letting it to perform a successful CAS violating the invariant.
68  * A counter in the 'next' offset (masked by WRAP_COUNTER) is used
69  * to prevent from this problem. It is incremented on wraparounds.
70  *
71  * The same ABA problem could also cause a stale 'ready' offset,
72  * which could be observed by the consumer. We set WRAP_LOCK_BIT in
73  * the 'seen' value before advancing the 'next' and clear this bit
74  * after the successful advancing; this ensures that only the stable
75  * 'ready' is observed by the consumer.
76  */

91 #define RBUF_OFF_MASK (0x00000000ffffffffUL)

105 struct ringbuf {
106  /* Ring buffer space. */
107  size_t space;
108 
109  /*
110  * The NEXT hand is atomically updated by the producer.
111  * WRAP_LOCK_BIT is set in case of wrap-around; in such case,
112  * the producer can update the 'end' offset.
113  */
114  volatile ringbuf_off_t next;
115  ringbuf_off_t end;
116 
117  /* The following are updated by the consumer. */
118  ringbuf_off_t written;
119  unsigned nworkers;
120  ringbuf_worker_t workers[];
121 };

123 /*
124  * ringbuf_setup: initialise a new ring buffer of a given length.
125  */
126 int
127 ringbuf_setup(ringbuf_t *rbuf, unsigned nworkers, size_t length)
128 {
129  if (length >= RBUF_OFF_MASK) {
130  errno = EINVAL;
131  return -1;
132  }
133  memset(rbuf, 0, offsetof(ringbuf_t, workers[nworkers]));
134  rbuf->space = length;
135  rbuf->end = RBUF_OFF_MAX;
136  rbuf->nworkers = nworkers;
137  return 0;
138 }

174 /*
175  * stable_nextoff: capture and return a stable value of the 'next' offset.
176  */
177 static inline ringbuf_off_t
178 stable_nextoff(ringbuf_t *rbuf)
179 {
180  unsigned count = SPINLOCK_BACKOFF_MIN;
181  ringbuf_off_t next;
182 retry:
183  next = atomic_load_explicit(&rbuf->next, memory_order_acquire);
184  if (next & WRAP_LOCK_BIT) {
185  SPINLOCK_BACKOFF(count);
186  goto retry;
187  }
188  ASSERT((next & RBUF_OFF_MASK) < rbuf->space);
189  return next;
190 }

192 /*
193  * stable_seenoff: capture and return a stable value of the 'seen' offset.
194  */
195 static inline ringbuf_off_t
196 stable_seenoff(ringbuf_worker_t *w)
197 {
198  unsigned count = SPINLOCK_BACKOFF_MIN;
199  ringbuf_off_t seen_off;
200 retry:
201  seen_off = atomic_load_explicit(&w->seen_off, memory_order_acquire);
202  if (seen_off & WRAP_LOCK_BIT) {
203  SPINLOCK_BACKOFF(count);
204  goto retry;
205  }
206  return seen_off;
207 }

209 /*
210  * ringbuf_acquire: request a space of a given length in the ring buffer.
211  *
212  * => On success: returns the offset at which the space is available.
213  * => On failure: returns -1.
214  */
215 ssize_t
216 ringbuf_acquire(ringbuf_t *rbuf, ringbuf_worker_t *w, size_t len)
217 {
218  ringbuf_off_t seen, next, target;
219 
220  ASSERT(len > 0 && len <= rbuf->space);
221  ASSERT(w->seen_off == RBUF_OFF_MAX);
222 
223  do {
224  ringbuf_off_t written;
225 
226  /*
227  * Get the stable 'next' offset. Save the observed 'next'
228  * value (i.e. the 'seen' offset), but mark the value as
229  * unstable (set WRAP_LOCK_BIT).
230  *
231  * Note: CAS will issue a memory_order_release for us and
232  * thus ensures that it reaches global visibility together
233  * with new 'next'.
234  */
235  seen = stable_nextoff(rbuf);
236  next = seen & RBUF_OFF_MASK;
237  ASSERT(next < rbuf->space);
238  atomic_store_explicit(&w->seen_off, next | WRAP_LOCK_BIT,
239  memory_order_relaxed);
240 
241  /*
242  * Compute the target offset. Key invariant: we cannot
243  * go beyond the WRITTEN offset or catch up with it.
244  */
245  target = next + len;
246  written = rbuf->written;
247  if (__predict_false(next < written && target >= written)) {
248  /* The producer must wait. */
249  atomic_store_explicit(&w->seen_off,
250  RBUF_OFF_MAX, memory_order_release);
251  return -1;
252  }
253 
254  if (__predict_false(target >= rbuf->space)) {
255  const bool exceed = target > rbuf->space;
256 
257  /*
258  * Wrap-around and start from the beginning.
259  *
260  * If we would exceed the buffer, then attempt to
261  * acquire the WRAP_LOCK_BIT and use the space in
262  * the beginning. If we used all space exactly to
263  * the end, then reset to 0.
264  *
265  * Check the invariant again.
266  */
267  target = exceed ? (WRAP_LOCK_BIT | len) : 0;
268  if ((target & RBUF_OFF_MASK) >= written) {
269  atomic_store_explicit(&w->seen_off,
270  RBUF_OFF_MAX, memory_order_release);
271  return -1;
272  }
273  /* Increment the wrap-around counter. */
274  target |= WRAP_INCR(seen & WRAP_COUNTER);
275  } else {
276  /* Preserve the wrap-around counter. */
277  target |= seen & WRAP_COUNTER;
278  }
279  } while (!atomic_compare_exchange_weak(&rbuf->next, &seen, target));
280 
281  /*
282  * Acquired the range. Clear WRAP_LOCK_BIT in the 'seen' value
283  * thus indicating that it is stable now.
284  *
285  * No need for memory_order_release, since CAS issued a fence.
286  */
287  atomic_store_explicit(&w->seen_off, w->seen_off & ~WRAP_LOCK_BIT,
288  memory_order_relaxed);
289 
290  /*
291  * If we set the WRAP_LOCK_BIT in the 'next' (because we exceed
292  * the remaining space and need to wrap-around), then save the
293  * 'end' offset and release the lock.
294  */
295  if (__predict_false(target & WRAP_LOCK_BIT)) {
296  /* Cannot wrap-around again if consumer did not catch-up. */
297  ASSERT(rbuf->written <= next);
298  ASSERT(rbuf->end == RBUF_OFF_MAX);
299  rbuf->end = next;
300  next = 0;
301 
302  /*
303  * Unlock: ensure the 'end' offset reaches global
304  * visibility before the lock is released.
305  */
306  atomic_store_explicit(&rbuf->next,
307  (target & ~WRAP_LOCK_BIT), memory_order_release);
308  }
309  ASSERT((target & RBUF_OFF_MASK) <= rbuf->space);
310  return (ssize_t)next;
311 }

326 /*
327  * ringbuf_consume: get a contiguous range which is ready to be consumed.
328  */
329 size_t
330 ringbuf_consume(ringbuf_t *rbuf, size_t *offset)
331 {
332  ringbuf_off_t written = rbuf->written, next, ready;
333  size_t towrite;
334 retry:
335  /*
336  * Get the stable 'next' offset. Note: stable_nextoff() issued
337  * a load memory barrier. The area between the 'written' offset
338  * and the 'next' offset will be the *preliminary* target buffer
339  * area to be consumed.
340  */
341  next = stable_nextoff(rbuf) & RBUF_OFF_MASK;
342  if (written == next) {
343  /* If producers did not advance, then nothing to do. */
344  return 0;
345  }
346 
347  /*
348  * Observe the 'ready' offset of each producer.
349  *
350  * At this point, some producer might have already triggered the
351  * wrap-around and some (or all) seen 'ready' values might be in
352  * the range between 0 and 'written'. We have to skip them.
353  */
354  ready = RBUF_OFF_MAX;
355 
356  for (unsigned i = 0; i < rbuf->nworkers; i++) {
357  ringbuf_worker_t *w = &rbuf->workers[i];
358  ringbuf_off_t seen_off;
359 
360  /*
361  * Skip if the worker has not registered.
362  *
363  * Get a stable 'seen' value. This is necessary since we
364  * want to discard the stale 'seen' values.
365  */
366  if (!atomic_load_explicit(&w->registered, memory_order_relaxed))
367  continue;
368  seen_off = stable_seenoff(w);
369 
370  /*
371  * Ignore the offsets after the possible wrap-around.
372  * We are interested in the smallest seen offset that is
373  * not behind the 'written' offset.
374  */
375  if (seen_off >= written) {
376  ready = MIN(seen_off, ready);
377  }
378  ASSERT(ready >= written);
379  }
380 
381  /*
382  * Finally, we need to determine whether wrap-around occurred
383  * and deduct the safe 'ready' offset.
384  */
385  if (next < written) {
386  const ringbuf_off_t end = MIN(rbuf->space, rbuf->end);
387 
388  /*
389  * Wrap-around case. Check for the cut off first.
390  *
391  * Reset the 'written' offset if it reached the end of
392  * the buffer or the 'end' offset (if set by a producer).
393  * However, we must check that the producer is actually
394  * done (the observed 'ready' offsets are clear).
395  */
396  if (ready == RBUF_OFF_MAX && written == end) {
397  /*
398  * Clear the 'end' offset if was set.
399  */
400  if (rbuf->end != RBUF_OFF_MAX) {
401  rbuf->end = RBUF_OFF_MAX;
402  }
403 
404  /*
405  * Wrap-around the consumer and start from zero.
406  */
407  written = 0;
408  atomic_store_explicit(&rbuf->written,
409  written, memory_order_release);
410  goto retry;
411  }
412 
413  /*
414  * We cannot wrap-around yet; there is data to consume at
415  * the end. The ready range is smallest of the observed
416  * 'ready' or the 'end' offset. If neither is set, then
417  * the actual end of the buffer.
418  */
419  ASSERT(ready > next);
420  ready = MIN(ready, end);
421  ASSERT(ready >= written);
422  } else {
423  /*
424  * Regular case. Up to the observed 'ready' (if set)
425  * or the 'next' offset.
426  */
427  ready = MIN(ready, next);
428  }
429  towrite = ready - written;
430  *offset = written;
431 
432  ASSERT(ready >= written);
433  ASSERT(towrite <= rbuf->space);
434  return towrite;
435 }

441 ringbuf_release(ringbuf_t *rbuf, size_t nbytes)
442 {
443  const size_t nwritten = rbuf->written + nbytes;
444 
445  ASSERT(rbuf->written <= rbuf->space);
446  ASSERT(rbuf->written <= rbuf->end);
447  ASSERT(nwritten <= rbuf->space);
448 
449  rbuf->written = (nwritten == rbuf->space) ? 0 : nwritten;
450 }
