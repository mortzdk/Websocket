738 static int read_internal(wss_server_t *server, wss_session_t *session, char *buffer) {
739  int n;
740 
741 #ifdef USE_OPENSSL
742  if ( NULL != server->ssl_ctx ) {
743  unsigned long err;
744 
745  n = SSL_read(session->ssl, buffer, server->config->size_buffer);
746  err = SSL_get_error(session->ssl, n);
747 
748  // There's no more to read from the kernel
749  if (err == SSL_ERROR_WANT_READ) {
750  n = 0;
751  } else
752 
753  // There's no space to write to the kernel, wait for filedescriptor.
754  if (err == SSL_ERROR_WANT_WRITE) {
755  WSS_log_debug("SSL_ERROR_WANT_WRITE");
756 
757  n = -2;
758  } else
759 
760  // Some error has occured.
761  if ( unlikely(err != SSL_ERROR_NONE && err != SSL_ERROR_ZERO_RETURN) ) {
762  char msg[1024];
763  while ( (err = ERR_get_error()) != 0 ) {
764  memset(msg, '\0', 1024);
765  ERR_error_string_n(err, msg, 1024);
766  WSS_log_error("SSL read failed: %s", msg);
767  }
768 
769  n = -1;
770  } else
771 
772  // If this was one of the error codes that we do accept, return 0
773  if ( unlikely(n < 0) ) {
774  n = 0;
775  }
776  } else {
777 #endif
778  do {
779  n = read(session->fd, buffer, server->config->size_buffer);
780  if (n == -1) {
781  if ( unlikely(errno == EINTR) ) {
782  errno = 0;
783  continue;
784  } else if ( unlikely(errno != EAGAIN && errno != EWOULDBLOCK) ) {
785  WSS_log_error("Read failed: %s", strerror(errno));
786  } else {
787  n = 0;
788  }
789  }
790  } while ( unlikely(0) );
791 #ifdef USE_OPENSSL
792  }
793 #endif
794 
795  return n;
796 }

805 static void handshake(wss_server_t *server, wss_session_t *session) {
806  int n;
807  wss_header_t *header;
808  wss_message_t *message;
809  enum HttpStatus_Code code;
810  char buffer[server->config->size_buffer];
811  bool ssl = false;
812 
813  memset(buffer, '\0', server->config->size_buffer);
814 
815 #ifdef USE_OPENSSL
816  if (NULL != server->ssl_ctx) {
817  ssl = true;
818  }
819 #endif
820 
821  WSS_log_trace("Preparing client header");
822 
823  if ( unlikely(NULL != session->header) ) {
824  header = session->header;
825  session->header = NULL;
826  } else {
827  if ( unlikely(NULL == (header = WSS_malloc(sizeof(wss_header_t)))) ) {
828  return;
829  }
830 
831  header->content = NULL;
832  header->method = NULL;
833  header->version = NULL;
834  header->path = NULL;
835  header->host = NULL;
836  header->payload = NULL;
837  header->length = 0;
838  header->ws_version = 0;
839  header->ws_type = UNKNOWN;
840  header->ws_protocol = NULL;
841  header->ws_upgrade = NULL;
842  header->ws_connection = NULL;
843  header->ws_extensions = NULL;
844  header->ws_extensions_count = 0;
845  header->ws_origin = NULL;
846  header->ws_key = NULL;
847  header->ws_key1 = NULL;
848  header->ws_key2 = NULL;
849  header->ws_key3 = NULL;
850  }
851 
852  WSS_log_trace("Reading headers");
853 
854  // Continue reading until we get no bytes back
855  do {
856  n = read_internal(server, session, buffer);
857 
858  switch (n) {
859  // Wait for IO for either read or write on the filedescriptor
860  case -2:
861  session->header = header;
862 
863  session->event = WRITE;
864 
865  return;
866  // An error occured, notify client by writing back to it
867  case -1:
868  WSS_log_trace("Rejecting HTTP request due to being unable to read from client");
869 
870  message = http_response(header, HttpStatus_InternalServerError, "Unable to read from client");
871  WSS_free_header(header);
872 
873  if ( likely(NULL != message && WSS_SUCCESS == write_internal(session, message)) ) {
874  session->state = WRITING;
875  session->event = READ;
876  WSS_write(server, session);
877  }
878 
879  return;
880  case 0:
881  break;
882  default:
883  // Reallocate space for the header and copy buffer into it
884  if ( unlikely(NULL == (header->content = WSS_realloc((void **) &header->content, header->length*sizeof(char), (header->length+n+1)*sizeof(char)))) ) {
885  WSS_log_fatal("Unable to realloc header content");
886  return;
887  }
888  memcpy(header->content+header->length, buffer, n);
889  header->length += n;
890  memset(buffer, '\0', server->config->size_buffer);
891 
892  // Check if payload from client is too large for the server to handle.
893  // If so write error back to the client
894  if ( unlikely(header->length > (server->config->size_header+server->config->size_uri+server->config->size_payload)) ) {
895  WSS_log_trace("Rejecting HTTP request as client payload is too large for the server to handle");
896 
897  message = http_response(header, HttpStatus_PayloadTooLarge,
898  "The given payload is too large for the server to handle");
899  WSS_free_header(header);
900 
901  if ( likely(NULL != message && WSS_SUCCESS == write_internal(session, message)) ) {
902  session->state = WRITING;
903  session->event = READ;
904  WSS_write(server, session);
905  }
906 
907  return;
908  }
909  break;
910  }
911  } while ( likely(n != 0) );
912 
913  WSS_log_debug("Client header: \n%s", header->content);
914 
915  WSS_log_trace("Starting parsing header received from client");
916 
917  // Parsing HTTP header
918  // If header could not be parsed correctly, notify client with response.
919  if ( (code = WSS_parse_header(session->fd, header, server->config)) != HttpStatus_OK ) {
920  WSS_log_trace("Rejecting HTTP request due to header not being correct");
921 
922  message = http_response(header, code, "Unable to parse header");
923  WSS_free_header(header);
924 
925  if ( likely(NULL != message && WSS_SUCCESS == write_internal(session, message)) ) {
926  session->state = WRITING;
927  session->event = READ;
928  WSS_write(server, session);
929  }
930 
931  return;
932  }
933 
934  // Serve favicon
935  if (strlen(header->path) == 12 && strncmp(header->path, "/favicon.ico", 12) == 0) {
936  message = favicon_response(header, code, server->config);
937  if (NULL != message) {
938  WSS_free_header(header);
939 
940  WSS_log_trace("Serving a favicon to the client");
941 
942  // Find and serve favicon.
943  if ( likely(WSS_SUCCESS == write_internal(session, message)) ) {
944  session->state = WRITING;
945  session->event = READ;
946  WSS_write(server, session);
947  }
948  } else {
949  WSS_log_trace("Rejecting HTTP request due to favicon not being available");
950 
951  // Else notify client that favicon could not be found
952  code = HttpStatus_NotFound;
953  message = http_response(header, code, "File not found");
954  WSS_free_header(header);
955 
956  if ( likely(NULL != message && WSS_SUCCESS == write_internal(session, message)) ) {
957  session->state = WRITING;
958  session->event = READ;
959  WSS_write(server, session);
960  }
961  }
962 
963  return;
964  }
965 
966  WSS_log_trace("Header successfully parsed");
967 
968  // Create Upgrade HTTP header based on clients header
969  code = WSS_upgrade_header(header, server->config, ssl, server->port);
970  switch (code) {
971  case HttpStatus_UpgradeRequired:
972  WSS_log_trace("Rejecting HTTP request as the service requires use of the Websocket protocol.");
973  message = upgrade_response(header, code,
974  "This service requires use of the Websocket protocol.");
975  break;
976  case HttpStatus_NotImplemented:
977  WSS_log_trace("Rejecting HTTP request as Websocket protocol is not yet implemented");
978  message = http_response(header, code,
979  "Websocket protocol is not yet implemented");
980  break;
981  case HttpStatus_SwitchingProtocols:
982  message = handshake_response(header, code);
983  break;
984  case HttpStatus_NotFound:
985  WSS_log_trace("Rejecting HTTP request as the page requested was not found.");
986  message = http_response(header, code,
987  "The page requested was not found.");
988  break;
989  case HttpStatus_Forbidden:
990  WSS_log_trace("Rejecting HTTP request as the origin is not allowed to establish a websocket connection.");
991  message = http_response(header, code,
992  "The origin is not allowed to establish a websocket connection.");
993  break;
994  default:
995  WSS_log_trace("Rejecting HTTP request as server was unable to parse http header as websocket request");
996  message = http_response(header, code,
997  "Unable to parse http header as websocket request");
998  break;
999  }
1000 
1001  // If code status isnt switching protocols, we notify client with a HTTP error
1002  if (code != HttpStatus_SwitchingProtocols) {
1003  WSS_log_trace("Unable to establish websocket connection");
1004 
1005  if ( likely(WSS_SUCCESS == write_internal(session, message)) ) {
1006  session->state = WRITING;
1007  session->event = READ;
1008  WSS_write(server, session);
1009  }
1010 
1011  WSS_free_header(header);
1012 
1013  return;
1014  }
1015 
1016  // Use echo protocol if none was chosen
1017  if (NULL == header->ws_protocol) {
1018  header->ws_protocol = WSS_find_subprotocol("echo");
1019  }
1020 
1021  // Notify websocket protocol of the connection
1022  header->ws_protocol->connect(session->fd, header->path, header->cookies);
1023 
1024  // Set session as fully handshaked
1025  session->handshaked = true;
1026  session->header = header;
1027 
1028  if ( likely(WSS_SUCCESS == write_internal(session, message)) ) {
1029  session->state = WRITING;
1030  session->event = READ;
1031  WSS_write(server, session);
1032  }
1033 }

1042 void WSS_read(wss_server_t *server, wss_session_t *session) {
1043  int n;
1044  size_t len;
1045  uint16_t code;
1046  wss_frame_t *frame;
1047  char *msg;
1048  bool closing = false;
1049  size_t offset = 0, prev_offset = 0;
1050  wss_frame_t **frames = NULL;
1051  char *payload = NULL;
1052  size_t payload_length = 0;
1053  size_t frames_length = 0;
1054  char *message;
1055  size_t message_length;
1056  wss_message_t *m;
1057  size_t i, j, k;
1058  size_t msg_length = 0;
1059  size_t msg_offset = 0;
1060  size_t starting_frame = 0;
1061  bool fragmented = false;
1062 
1063  // If no initial header has been seen for the session, the websocket
1064  // handshake is yet to be made.
1065  if ( unlikely(! session->handshaked) ){
1066  WSS_log_trace("Doing websocket handshake");
1067  handshake(server, session);
1068  return;
1069  }
1070 
1071  WSS_log_trace("Starting initial steps to read from client");
1072 
1073  char buffer[server->config->size_buffer];
1074  memset(buffer, '\0', server->config->size_buffer);
1075 
1076  // Use earlier payload
1077  payload = session->payload;
1078  payload_length = session->payload_length;
1079  offset = session->offset;
1080  frames = session->frames;
1081  frames_length = session->frames_length;
1082  session->payload = NULL;
1083  session->payload_length = 0;
1084  session->offset = 0;
1085  session->frames = NULL;
1086  session->frames_length = 0;
1087 
1088  // If handshake has been made, we can read the websocket frames from
1089  // the connection
1090  do {
1091  n = read_internal(server, session, buffer);
1092 
1093  switch (n) {
1094  // Wait for IO for either read or write on the filedescriptor
1095  case -2:
1096  WSS_log_trace("Detected that server needs further IO to complete the reading");
1097  session->payload = payload;
1098  session->payload_length = payload_length;
1099 
1100  session->event = WRITE;
1101 
1102  return;
1103  // An error occured
1104  case -1:
1105  WSS_free((void **) &payload);
1106 
1107  frame = WSS_closing_frame(CLOSE_UNEXPECTED, NULL);
1108  message_length = WSS_stringify_frame(frame, &message);
1109  WSS_free_frame(frame);
1110 
1111  if ( unlikely(NULL == (m = WSS_malloc(sizeof(wss_message_t)))) ) {
1112  WSS_log_error("Unable to allocate the message structure");
1113  WSS_free((void **) &message);
1114  session->closing = true;
1115  return;
1116  }
1117  m->msg = message;
1118  m->length = message_length;
1119  m->framed = true;
1120 
1121  if ( likely(WSS_SUCCESS == write_internal(session, m)) ) {
1122  session->state = WRITING;
1123  session->event = READ;
1124  WSS_write(server, session);
1125  }
1126  return;
1127  // No new data received
1128  case 0:
1129  break;
1130  // Empty buffer into payload
1131  default:
1132  if ( unlikely(NULL == (payload = WSS_realloc((void **) &payload, payload_length*sizeof(char), (payload_length+n+1)*sizeof(char)))) ) {
1133  WSS_log_error("Unable to reallocate the payload");
1134  session->closing = true;
1135  return;
1136  }
1137 
1138  memcpy(payload+payload_length, buffer, n);
1139  payload_length += n;
1140  memset(buffer, '\0', server->config->size_buffer);
1141 
1142  }
1143  } while ( likely(n != 0) );
1144 
1145  WSS_log_trace("Payload from client was read. Continues flow by parsing frames.");
1146 
1147  // Parse the payload into websocket frames
1148  do {
1149  prev_offset = offset;
1150 
1151  if ( unlikely(NULL == (frame = WSS_parse_frame(payload, payload_length, &offset))) ) {
1152  WSS_log_trace("Unable to parse frame");
1153  WSS_free((void **) &payload);
1154  session->closing = true;
1155  return;
1156  }
1157 
1158  // Check if we were forced to read beyond the payload to create a full frame
1159  if ( unlikely(offset > payload_length) ) {
1160  WSS_log_trace("Detected that data was missing in order to complete frame, will wait for more");
1161 
1162  WSS_free_frame(frame);
1163 
1164  session->payload = payload;
1165  session->payload_length = payload_length;
1166  session->offset = prev_offset;
1167  session->frames = frames;
1168  session->frames_length = frames_length;
1169  session->event = READ;
1170 
1171  return;
1172  }
1173 
1174  // If no extension is negotiated, the rsv bits must not be used
1175  if ( unlikely(NULL == session->header->ws_extensions && (frame->rsv1 || frame->rsv2 || frame->rsv3)) ) {
1176  WSS_log_trace("Protocol Error: rsv bits must not be set without using extensions");
1177  WSS_free_frame(frame);
1178  frame = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1179  } else
1180 
1181  // If opcode is unknown
1182  if ( unlikely((frame->opcode >= 0x3 && frame->opcode <= 0x7) ||
1183  (frame->opcode >= 0xB && frame->opcode <= 0xF)) ) {
1184  WSS_log_trace("Type Error: Unknown upcode");
1185  WSS_free_frame(frame);
1186  frame = WSS_closing_frame(CLOSE_TYPE, NULL);
1187  } else
1188 
1189  // Server expects all received data to be masked
1190  if ( unlikely(! frame->mask) ) {
1191  WSS_log_trace("Protocol Error: Client message should always be masked");
1192  WSS_free_frame(frame);
1193  frame = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1194  } else
1195 
1196  // Control frames cannot be fragmented
1197  if ( unlikely(! frame->fin && frame->opcode >= 0x8 && frame->opcode <= 0xA) ) {
1198  WSS_log_trace("Protocol Error: Control frames cannot be fragmented");
1199  WSS_free_frame(frame);
1200  frame = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1201  } else
1202 
1203  // Check that frame is not too large
1204  if ( unlikely(frame->payloadLength > server->config->size_frame) ) {
1205  WSS_log_trace("Protocol Error: Control frames cannot be fragmented");
1206  WSS_free_frame(frame);
1207  frame = WSS_closing_frame(CLOSE_BIG, NULL);
1208  } else
1209 
1210  // Control frames cannot have a payload length larger than 125 bytes
1211  if ( unlikely(frame->opcode >= 0x8 && frame->opcode <= 0xA && frame->payloadLength > 125) ) {
1212  WSS_log_trace("Protocol Error: Control frames cannot have payload larger than 125 bytes");
1213  WSS_free_frame(frame);
1214  frame = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1215  } else
1216 
1217  // Close frame
1218  if ( unlikely(frame->opcode == CLOSE_FRAME) ) {
1219  // A code of 2 byte must be present if there is any application data for closing frame
1220  if ( unlikely(frame->applicationDataLength > 0 && frame->applicationDataLength < 2) ) {
1221  WSS_log_trace("Protocol Error: Closing frame with payload too small bytewise error code");
1222  WSS_free_frame(frame);
1223  frame = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1224  } else
1225 
1226 
1227  // The payload after the code, must be valid UTF8
1228  if ( unlikely(frame->applicationDataLength >= 2 && ! utf8_check(frame->payload+2, frame->applicationDataLength-2)) ) {
1229  WSS_log_trace("Protocol Error: Payload of error frame must be valid UTF8.");
1230  WSS_free_frame(frame);
1231  frame = WSS_closing_frame(CLOSE_UTF8, NULL);
1232  } else
1233 
1234  // Check status code is within valid range
1235  if (frame->applicationDataLength >= 2) {
1236  // Copy code
1237  memcpy(&code, frame->payload, sizeof(uint16_t));
1238  code = ntohs(code);
1239 
1240  WSS_log_debug("Closing frame code: %d", code);
1241 
1242  // Current rfc6455 codes
1243  if ( unlikely(code < 1000 || (code >= 1004 && code <= 1006) || (code >= 1015 && code < 3000) || code >= 5000) ) {
1244  WSS_log_trace("Protocol Error: Closing frame has invalid error code");
1245  WSS_free_frame(frame);
1246  frame = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1247  }
1248  }
1249  } else
1250 
1251  // Pong
1252  if ( unlikely(frame->opcode == PONG_FRAME) ) {
1253  WSS_log_trace("Pong received");
1254 
1255  if ( session->pong == NULL || (session->pong_length == frame->applicationDataLength &&
1256  memcmp(session->pong, frame->payload+frame->extensionDataLength, frame->applicationDataLength) == 0) ) {
1257  clock_gettime(CLOCK_MONOTONIC, &session->alive);
1258  }
1259 
1260  WSS_free_frame(frame);
1261 
1262  continue;
1263  } else
1264 
1265  // Ping
1266  if ( unlikely(frame->opcode == PING_FRAME) ) {
1267  WSS_log_trace("Ping received");
1268  frame = WSS_pong_frame(frame);
1269  }
1270 
1271  if ( unlikely(NULL == (frames = WSS_realloc((void **) &frames, frames_length*sizeof(wss_frame_t *),
1272  (frames_length+1)*sizeof(wss_frame_t *)))) ) {
1273  WSS_log_error("Unable to reallocate frames");
1274  session->closing = true;
1275  return;
1276  }
1277  frames[frames_length] = frame;
1278  frames_length += 1;
1279 
1280  // Check if frame count is exceeded
1281  if ( unlikely(frames_length > server->config->max_frames) ) {
1282  WSS_free_frame(frame);
1283  frame = WSS_closing_frame(CLOSE_BIG, NULL);
1284  frames[frames_length-1] = frame;
1285  }
1286 
1287  // Close
1288  if ( unlikely(frame->opcode == CLOSE_FRAME) ) {
1289  closing = true;
1290  WSS_log_trace("Stopping frame validation as closing frame was parsed");
1291  break;
1292  }
1293  } while ( likely(offset < payload_length) );
1294 
1295  WSS_free((void **) &payload);
1296 
1297  WSS_log_trace("A total of %lu frames was parsed.", frames_length);
1298 
1299  WSS_log_trace("Starting frame validation");
1300 
1301  // Validating frames.
1302  for (i = 0; likely(i < frames_length); i++) {
1303  // If we are not processing a fragmented set of frames, expect the
1304  // opcode different from the continuation frame.
1305  if ( unlikely(! fragmented && (frames[i]->opcode == CONTINUATION_FRAME)) ) {
1306  WSS_log_trace("Protocol Error: continuation opcode used in non-fragmented message");
1307  for (j = i; likely(j < frames_length); j++) {
1308  WSS_free_frame(frames[j]);
1309  }
1310  frames[i] = WSS_closing_frame(CLOSE_PROTOCOL, NULL);
1311  frames_length = i+1;
1312  closing = true;
1313  break;
1314  }
1315 
1316  // If we receive control frame within fragmented frames.
1317  if ( unlikely(fragmented && (frames[i]->opcode >= 0x8 && frames[i]->opcode <= 0xA)) ) {
1318  WSS_log_trace("Received control frame within fragmented message");
1319 
1320  frame = frames[i];
1321  // If we received a closing frame substitue the fragment with the
1322  // closing frame and end validation
1323  if (frames[i]->opcode == CLOSE_FRAME) {
1324  WSS_log_trace("Stopping further validation of fragmented message as a closing frame was detected");
1325 
1326  for (j = starting_frame; likely(j < frames_length); j++) {
1327  if ( unlikely(j != i) ) {
1328  WSS_free_frame(frames[j]);
1329  } else {
1330  frames[j] = NULL;
1331  }
1332  }
1333  frames[starting_frame] = frame;
1334  frames_length = starting_frame+1;
1335  fragmented = false;
1336  closing = true;
1337  break;
1338  // Else rearrange the frames, such that control frame is first
1339  } else {
1340  WSS_log_trace("Rearranging frames such that control frame will be written before fragmented frames");
1341  for (j = i; likely(j > starting_frame); j--) {
1342  frames[j] = frames[j-1];
1343  }
1344  frames[starting_frame] = frame;
1345  starting_frame++;
1346  }
1347  } else
